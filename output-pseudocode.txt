# import the canvas implementation from TA
from tacanvas import TurtleGraphics

# instantiate either a real TurtleArtWindow instance 
# or an instance of a dummy sub-class that supports only the stuff 
# TurtleGraphics needs
window = TurtleArtWindow(gtk.DrawingArea(), ...)

# setup the canvas to draw on
canvas = TurtleGraphics(window, gtk.screen_width(), gtk.screen_height())



# auto-converted block code
# example: draw a square
for i in range(4):
    canvas.forward(100)
    canvas.right(90)
